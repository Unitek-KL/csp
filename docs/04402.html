RSTC_RESET_CAUSE Enumeration

Reset cause 

This enumeration provides identifiers for General, Backup, Watchdog, Software, User reset, CPU clock failure and 32.768 kHz crystal failure causes.

C
typedef enum {
  RSTC_GENERAL_RESET = RSTC_SR_RSTTYP_GENERAL_RST,
  RSTC_WAKEUP_RESET = RSTC_SR_RSTTYP_BACKUP_RST,
  RSTC_WATCHDOG_RESET = RSTC_SR_RSTTYP_WDT_RST,
  RSTC_SOFTWARE_RESET = RSTC_SR_RSTTYP_SOFT_RST,
  RSTC_USER_RESET = RSTC_SR_RSTTYP_USER_RST,
  RSTC_CPU_FAIL_RESET = RSTC_SR_RSTTYP_CPU_FAIL_RST,
  RSTC_SLCK_XTAL_RESET = RSTC_SR_RSTTYP_SLCK_XTAL_RST
} RSTC_RESET_CAUSE;
