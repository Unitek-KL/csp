RTC_ALARM_MASK Enumeration

RTCC Alaram Mask 

This enumeration defines the possible options for recurrence of RTCC alarm. Used as the second parameter for RTCC_AlarmSet() function.

C
typedef enum {
  RTC_ALARM_MASK_HALF_SECOND,
  RTC_ALARM_MASK_SECOND,
  RTC_ALARM_MASK_10_SECONDS,
  RTC_ALARM_MASK_SS,
  RTC_ALARM_MASK_10_MINUTES,
  RTC_ALARM_MASK_HOUR,
  RTC_ALARM_MASK_HHMISS,
  RTC_ALARM_MASK_WEEK,
  RTC_ALARM_MASK_MONTH,
  RTC_ALARM_MASK_YEAR,
  RTC_ALARM_MASK_OFF
} RTC_ALARM_MASK;
